===========================================
COMMANDES DE DÉPLOIEMENT - À EXÉCUTER SUR LE SERVEUR
===========================================

# ÉTAPE 1 : Mise à jour du système
apt update && apt upgrade -y

# ÉTAPE 2 : Installation de Node.js 18
apt install -y curl
curl -fsSL https://deb.nodesource.com/setup_18.x | bash -
apt install -y nodejs
node --version
npm --version

# ÉTAPE 3 : Installation de Git
apt install -y git

# ÉTAPE 4 : Installation de Nginx
apt install -y nginx
systemctl start nginx
systemctl enable nginx

# ÉTAPE 5 : Installation de PM2
npm install -g pm2

# ÉTAPE 6 : Cloner le projet
mkdir -p /var/www
cd /var/www
git clone https://github.com/DevMick/sunrefci.git
cd sunrefci
npm install

# ÉTAPE 7 : Build de l'application
npm run build

# ÉTAPE 8 : Démarrer avec PM2
pm2 start npm --name "sunref-ci" -- start
pm2 save
pm2 startup
# ⚠️ IMPORTANT : Exécuter la commande affichée par pm2 startup

# ÉTAPE 9 : Configuration Nginx
cat > /etc/nginx/sites-available/sunrefcotedivoire.com << 'EOF'
server {
    listen 80;
    server_name sunrefcotedivoire.com www.sunrefcotedivoire.com;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
EOF

ln -s /etc/nginx/sites-available/sunrefcotedivoire.com /etc/nginx/sites-enabled/
nginx -t
systemctl reload nginx

# ÉTAPE 10 : Configuration du pare-feu
apt install -y ufw
ufw allow OpenSSH
ufw allow 'Nginx Full'
ufw enable

# ÉTAPE 11 : Installation SSL (Let's Encrypt)
apt install -y certbot python3-certbot-nginx
certbot --nginx -d sunrefcotedivoire.com -d www.sunrefcotedivoire.com

# ÉTAPE 12 : Vérification
pm2 status
pm2 logs sunref-ci --lines 50
